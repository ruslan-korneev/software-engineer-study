### try-except
```python
def safe_int(value, default=0):
    try:
        return int(value)
    except (ValueError, TypeError):
        return default

safe_int("42")       # 42
safe_int("hello")    # 0
safe_int(None)       # 0
safe_int("3.14")     # 0

def safe_float(value, default=0.0):
    try:
        return float(value)
    except (ValueError, TypeError):
        return default
```
